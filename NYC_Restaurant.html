Scatterplot for Grade-A Japanese restaurants distribution in NYC

nyc_inspections %>% 
  filter(!is.na(building),
         grade == "A",
         cuisine_description == "Japanese") %>%
  mutate(text_label = str_c("Grade: ", grade, '\nScore: ', score)) %>% 
  plot_ly(
    x = ~longitude, y = ~latitude, type = "scatter", mode = "markers",
    color = ~boro, text = ~text_label, alpha = 0.5, colors = "Set2")

Barchart for numbers of Grade-A Japanese restaurants in different rgion

nyc_inspections %>% 
  filter(!is.na(building),
         grade == "A",
         cuisine_description == "Japanese") %>% 
  group_by(boro) %>%
  summarize(number = n()) %>% 
  plot_ly(x = ~boro, y = ~number, color = ~boro, type = "bar",
          colors = "Set2")
